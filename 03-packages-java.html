Java Packages

Java Packages
  • Posted By Cristopher André
  • 1 April 2019
  • JAVA - Java Packages

A package in Java is used to group related classes. Think of it as a folder in a file directory. We use packages to avoid name conflicts, and to write a better maintainable code


-> Two or more imports with the same same name doens't compile.

-> To use two or more objects with the same name we must use the Fully Qualified Name.

-> Default package classes never can be accessed.

-> Imports with ".*" wont import sub-packages.

-> You can use import static to usi static variables or methods

Example:

                
                      import com.javaCertification.packages.first.Address;
                      import com.javaCertification.packages.first.Person;

                      // Instead of declaring each import separate, we can import the entire package.
                      import com.javaCertification.packages.first.*;

                      // Can't import two classes with the same name from different packages
                      //import java.util.Date;
                      //import java.sql.Date;

                      // In this case the Java will import the java.sql.Date
                      import java.util.*;
                      import java.sql.Date;

                      import com.javaCertification.packages.third.*;

                      // Static Import you can import variable or method to be accessed directly
                      import static com.javaCertification.packages.third.Util.PI;
                      import static com.javaCertification.packages.third.Util.bigger;
                      // Import all statics from package
                      import static com.javaCertification.packages.third.Util.*;


                      /*
                      * Two or more imports with the same same name doens't compile.
                      * To use two or more objects with the same name we must use the Fully Qualified Name.
                      * Default package classes never can be accessed.
                      * Imports with ".*" wont import sub-packages.
                      * You can use import static to usi static variables or methods 
                      * 
                      */

                      public class World {
                        
                        // To reference a class from another package, we use package + className
                        // The class referenced must be Public to be imported.	
                        com.javaCertification.packages.first.Person person;

                        // It's not a good practice to use the Fully Qualified Name
                        public com.javaCertification.packages.first.Person getPerson() {
                          return person;
                        }
                        
                        // Use the import so you don't have to type the Fully Qualified Name
                        Person person2;
                        public Person getPerson2() {
                          return person2;
                        }	
                        
                        // Reference of another object
                        Address address;
                        
                        // This code wont work, because you're trying to instance a java.util.Date() in the type java.sql.Date
                        //Date date = new java.util.Date();
                        
                        // If you want to use both Dates you'll need to use the Fully Qualified name in both
                        java.util.Date dateUtil = new java.util.Date();
                        java.sql.Date dateSql = new java.sql.Date(1220227200);
                        
                        public void doSomething() {
                          // Using Util import
                          double pi = Util.PI;
                          int bigger = Util.bigger(3, 4);
                          
                          // Using static import
                          double pi2 = PI;
                          int bigger2 = bigger(3, 4);
                        }

                      }
                      
                    
                      package com.javaCertification.packages.first;

                      import com.javaCertification.packages.first.Address; // Optional because they are in the same package

                      public class Person {
                        
                        String name;
                        
                        // Address is in the same package than Person, so the import isn't necessary
                        Address address;

                      }

                    
                      package com.javaCertification.packages.first;

                      public class Address {
                        
                        String street;
                        String number;
                        String neighborhood;

                      }

                    
                      package com.javaCertification.packages.third;

                      public class Util {
                        
                        public static double PI = 3.14;
                        
                        public static int bigger(int a, int b) {
                          if(a > b) {
                            return a;
                          }else {
                            return b;
                          }
                        }

                      }
